// ======================================== // AirGuard ESP8266 Sender — NodeMCU // Sensors: MQ-2, DHT11, Flame sensor // ======================================== #include <ESP8266WiFi.h> #include <ESP8266HTTPClient.h> #include <ArduinoJson.h> #include <DHT.h> const char* ssid = "YOUR_WIFI_SSID"; const char* password = "YOUR_WIFI_PASSWORD"; const char* serverURL = "http://YOUR_SERVER_IP:3000/api/data"; #define DHT_PIN D4 // DHT11 data pin #define DHT_TYPE DHT11 #define MQ2_PIN A0 // MQ-2 analog output #define FLAME_PIN D5 // Flame sensor digital out DHT dht(DHT_PIN, DHT_TYPE); void setup() { Serial.begin(115200); dht.begin(); pinMode(FLAME_PIN, INPUT); WiFi.begin(ssid, password); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\nConnected: " + WiFi.localIP().toString()); } void loop() { float temp = dht.readTemperature(); float humidity = dht.readHumidity(); int gasLevel = analogRead(MQ2_PIN); // 0–1023 bool flame = !digitalRead(FLAME_PIN); // LOW = fire if (!isnan(temp) && WiFi.status() == WL_CONNECTED) { HTTPClient http; WiFiClient client; http.begin(client, serverURL); http.addHeader("Content-Type", "application/json"); StaticJsonDocument<128> doc; doc["gas"] = gasLevel; doc["flame"] = flame; doc["temp"] = temp; doc["humidity"] = humidity; doc["device"] = "ESP8266-01"; String payload; serializeJson(doc, payload); int code = http.POST(payload); Serial.printf("POST %s → %d\n", serverURL, code); http.end(); } delay(3000); // Send every 3 seconds }
// POST /api/data — request body { "gas": 342, // MQ-2 raw ADC 0–1023 "flame": false, // true = fire detected "temp": 28.4, // °C from DHT11 "humidity": 65.2, // % from DHT11 "device": "ESP8266-01" // device identifier } // GET /api/latest — response { "gas": 342, "flame": false, "temp": 28.4, "humidity": 65.2, "timestamp": "2025-04-10T14:32:00Z", "aqi": "SAFE" } // WebSocket push message (every new reading) { "type": "update", "data": { "gas": 342, ... } }
const express = require('express'); const { WebSocketServer } = require('ws'); const cors = require('cors'); const app = express(); const PORT = 3000; app.use(cors()); app.use(express.json()); app.use(express.static('public')); // serve dashboard HTML let latest = {}; let history = []; const clients = new Set(); const getAQI = (gas) => gas < 300 ? 'SAFE' : gas < 700 ? 'MODERATE' : 'DANGEROUS'; app.post('/api/data', (req, res) => { const data = { ...req.body, timestamp: new Date().toISOString(), aqi: getAQI(req.body.gas) }; latest = data; history.push(data); if (history.length > 200) history.shift(); // Broadcast to all WebSocket clients const msg = JSON.stringify({ type: 'update', data }); clients.forEach(ws => ws.readyState === 1 && ws.send(msg)); res.json({ ok: true }); }); app.get('/api/latest', (req, res) => res.json(latest)); app.get('/api/history', (req, res) => res.json(history.slice(-100))); const server = app.listen(PORT, () => console.log(`AirGuard server → http://localhost:${PORT}`)); const wss = new WebSocketServer({ server, path: '/ws' }); wss.on('connection', (ws) => { clients.add(ws); ws.send(JSON.stringify({ type: 'init', data: latest })); ws.on('close', () => clients.delete(ws)); });
